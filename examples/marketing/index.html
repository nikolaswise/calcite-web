Image switchers for switching images.

Image Switcher

<!-- Sample HTML -->
<div class="grid-container">
  <div class="column-10 pre-1 post-1 padding-leader-6">
    <a href="#" class="image-toggle js-view-toggle is-active" data-toggle-set="image-switcher" data-toggle="foo">
      <img src="/assets/img/illustrations/01-circle.svg" alt="" class="image-toggle-preview">
      <span class="image-toggle-description">Image One</span>
    </a>
    <a href="#" class="image-toggle js-view-toggle" data-toggle-set="image-switcher" data-toggle="bar">
      <img src="/assets/img/illustrations/02-circle.svg" alt="" class="image-toggle-preview">
      <span class="image-toggle-description">Image Two</span>
    </a>
    <a href="#" class="image-toggle js-view-toggle" data-toggle-set="image-switcher" data-toggle="baz">
      <img src="/assets/img/illustrations/03-circle.svg" alt="" class="image-toggle-preview">
      <span class="image-toggle-description"> Image Three</span>
    </a>
  </div>
  <div class="column-12">
    <img class="image-panel is-active" src="/assets/img/illustrations/01-square.svg" alt="one" data-toggle-set="image-switcher" data-toggle="foo">
    <img class="image-panel" src="/assets/img/illustrations/02-square.svg" alt="two" data-toggle-set="image-switcher" data-toggle="bar">
    <img class="image-panel" src="/assets/img/illustrations/03-square.svg" alt="three" data-toggle-set="image-switcher" data-toggle="baz">
  </div>
</div>

<div class="carousel">
  <div class="carousel-panel is-active" data-toggle-set="carousel" data-toggle="foo">

  </div>
  <div class="carousel-panel" data-toggle-set="carousel" data-toggle="baz">
    <img src="/assets/img/illustrations/02-picture.svg" alt="two">
  </div>
  <div class="carousel-panel" data-toggle-set="carousel" data-toggle="bar">
    <img src="/assets/img/illustrations/03-picture.svg" alt="three">
  </div>
</div>

<nav class="carousel-nav">
